%% handy cntrbox command
%%
%
% box #1 by #2 (centered on the line 5pt above the current baseline)
% containing object #3
%
\def\cntrbox#1#2#3{%
\newdimen\htobj\newdimen\htbx%
\makebox[#1]{\htbx=#2%
\setbox1=\hbox{#3}%
\htobj=\ht1%
\divide\htbx\tw@%
\divide\htobj\tw@%
\advance\htbx 5pt%
\advance\htobj -5pt%
\rule[-\htbx]{0mm}{#2}%
\lower \htobj \box1%
}}
