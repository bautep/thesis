\macrodef{\def\onggt{%
   \setbox1=\hbox{$\sim$} \dimen1=\wd1         % get size of ~
        \rlap{\hbox to \dimen1{\hfil\raise.08em\hbox{$>$}\hfil}}   % center >
      \lower.48em\hbox{$\sim$}                                   % and print ~
        \raise.36em\hbox{}
}}
	{$\onggt$ }{greater than or of the order of}
\macrodef{\def\onglt{%
   \setbox1=\hbox{$\sim$} \dimen1=\wd1         % get size of ~
        \rlap{\hbox to \dimen1{\hfil\raise.08em\hbox{$<$}\hfil}}   % center >
      \lower.48em\hbox{$\sim$}                                   % and print ~
        \raise.36em\hbox{}
}}
	{$\onglt$ }{less than or of the order of}
